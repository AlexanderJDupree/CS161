function size() { // lists the size of all files and total size of directory
	du -h $1 | sort -n
	du -hcs $1
}

export -f size

function up() { // goes up n directories
	LIMIT=$1
	P=$PWD
	for ((i=1; i <= LIMIT; i++))
	do 
		P=$P/..
	done
	cd $P
}
